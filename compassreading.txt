void compassreading()
{
  Wire.beginTransmission(HMC6352SlaveAddress);
  Wire.write(HMC6352ReadAddress);              //"Get Data" command 
  Wire.endTransmission();

  delay(6); //time delays required between receipt of the command and data processing : 6ms, stated on datasheet 

  Wire.requestFrom(HMC6352SlaveAddress, 2); //get MSB (Most Significant Byte) and LSB (Least Significant Byte)

  byte MSB = Wire.read(); // byte representing X axis
  byte LSB = Wire.read(); // byte representing Y axis

  float headingSum = (MSB << 8) + LSB; //(MSB/LSB sum, i.e. sum of X axis and Y axis )
  float headingInt = headingSum / 10;  //"The heading output data will be the value in tenths of degrees from zero to 3599", hence dividing by ten

  Serial.println("degrees:");
  Serial.print(headingInt);
}
